def detect_sudden_motion(frame):
    global prev_gray
    gray = cv2.cvtColor(frame, cv2.COLOR_BGR2GRAY)
    gray = cv2.GaussianBlur(gray, (21, 21), 0)
    if prev_gray is None:
        prev_gray = gray
        return False, 0
    flow = cv2.calcOpticalFlowFarneback(prev_gray, gray, None, 0.5, 3, 15, 3, 5, 1.2, 0)
    prev_gray = gray
    vx = flow[..., 0]
    vy = flow[..., 1]
    magnitude = np.sqrt(vx**2 + vy**2)
    mean_energy = np.mean(magnitude)
    max_energy = np.max(magnitude)
    downward = np.mean(vy[vy > 0]) if np.any(vy > 0) else 0
    if mean_energy > 3.0 and max_energy > 15.0 and downward > 2.0:
        return True, mean_energy
    return False, mean_energy

        # ── STATE MACHINE ──────────────────────────────────────────
